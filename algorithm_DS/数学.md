# 1 快速幂

```cpp
long long pow(long long x, int n) {
    long long res = 1;
    for (; n; n /= 2) {
        if (n % 2) {
            res = res * x % MOD;
        }
        x = x * x % MOD;
    }
    return res;
}
```

## 预处理回文数

```cpp
//生成回文数：pal[i] <= 1e9 , pal为有序数组
vector<int> pal;
auto init = [] {
    for (int base = 1; base <= 10000; base *= 10) {
        // 生成奇数长度回文数
        for (int i = base; i < base * 10; i++) {
            int x = i;
            for (int t = i / 10; t; t /= 10) {   //向x拼接t
                x = x * 10 + t % 10;
            }
            pal.push_back(x);
        }
        // 生成偶数长度回文数
        if (base <= 1000) {
            for (int i = base; i < base * 10; i++) {
                int x = i;
                for (int t = i; t; t /= 10) {  //向x拼接t
                    x = x * 10 + t % 10;
                }
                pal.push_back(x);
            }
        }
    }
    pal.push_back(1'000'000'001); // 哨兵，防止下面代码中的 i 下标越界
    return 0;
}();
```

# 4 质数筛

##  4.1 埃氏筛

```cpp
const int MX = 1e6;
vector<int> primes;

int init = []() {
    bool np[MX + 1]{};	//质数false， 非质数true;
    np[1] = true;
    for (int i = 2; i <= MX; i++) {
		if (!np[i]) {
            primes.push_back(i);
            for (int j = i * i; j <= MX; j += i) {
                np[j] = true;
            }
        }        
    }
}();
```

## 4.2 欧拉筛

```cpp
const int MX = 1e6;
vector<int> primes;

int init = [](){
    bool np[MX + 1]{};	//质数false， 非质数true;
    for (int i = 2; i <= MX; i++) {
        if (!np[i]) primes.push_back(i);
        for (auto x : primes) {
            if (x * i > MX) break;
            np[x * i] = true;
            if (i % x == 0) break;
        }
    }
}();
```
